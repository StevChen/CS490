function request(req, callback){
  	var url = "https://web.njit.edu/~lhc2/InstructorPage/upload.php";
	var xhttp = new XMLHttpRequest();

	xhttp.open("POST", url, false);
	xhttp.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
	xhttp.onreadystatechange = function() {
    if (this.readyState == 4 && this.status == 200) {
      console.log(this);
      callback(this.responseText);
    }
	};
	var jsonFormat = JSON.stringify(req);
	xhttp.send(jsonFormat);
}

function parseAns(text){
	var parsedText = text.replace(/<nl>/g,"\n");
	parsedText = parsedText.replace(/<dq>/g, '\"');
	return parsedText;
}

function buildQuestion(json){
	var text = "Please write a function named " + json.name + " with arguments " + json.args + " that : " + json.txt + "\n";
	//text += "Example: " + json.output +"\n";
	return text;
}

function setGradeAndComment(){
	var newGrade = this.parentNode.childNodes[0].value;
	//console.log(newGrade);
	var newComment = this.parentNode.childNodes[1].value;
	//console.log(newComment);
	if(newGrade.value != ""){
		var stdRequest = {
			act: "setAnswerGrade",
			id: this.parentNode.parentNode.id,
			exam: examid,
			user: sessionStorage.getItem("stud"),
			grade: newGrade
		}
		console.log(stdRequest);
		request(stdRequest, setGradeSuccess);
	}
	if(newComment.value != ""){
		var stdRequest = {
			act: "setAnswerComment",
			id: this.parentNode.parentNode.id,
			exam: examid,
			user: sessionStorage.getItem("stud"),
			comment: newComment
		}
		console.log(stdRequest);
		request(stdRequest, setCommentSuccess);
	}
}
